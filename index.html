            const gameOfLife = (str) => {
              let matrix = str.split('_');
              let nextGeneration = '';
                
              for (let row = 0; row < matrix.length; row++) {
                //checking the passed NxM grid
                if (matrix.length !== matrix[row].length) {
                  return str.replaceAll('1', '0');
                }
                  
                for (let col = 0; col < matrix[row].length; col++) {
                  let neighboorCount = 0;
                    
                  neighboorCount += +(matrix[row - 1]?.[col - 1] ?? 0); //top left neighboor
                  neighboorCount += +(matrix[row - 1]?.[col] ?? 0); //top neighboor
                  neighboorCount += +(matrix[row - 1]?.[col + 1] ?? 0); //top right neighboor
                  neighboorCount += +(matrix[row][col - 1] ?? 0); //left neighboor
                  neighboorCount += +(matrix[row][col + 1] ?? 0); //right neighboor
                  neighboorCount += +(matrix[row + 1]?.[col - 1] ?? 0); //buttom left neighboor
                  neighboorCount += +(matrix[row + 1]?.[col] ?? 0); //buttom neighboor
                  neighboorCount += +(matrix[row + 1]?.[col + 1] ?? 0); //buttom right neighboor
                  
                  if (neighboorCount === 2 || neighboorCount === 3) {
                    nextGeneration += '1';
                  } else {
                    nextGeneration += '0';
                  }
                }
                
                //adding a separator after each row
                str.length !== nextGeneration.length && (nextGeneration += '_');
              }
            
              return nextGeneration;
            };
